## 代码块展示

```javascript
function createDarkTheme(name) {
  return {
    name,
    colors: {
      primary: '#10B981',
      headingBg: '#23e89a',
      codeBg: '#0F172A'
    },
    apply() {
      console.log(`Theme "${this.name}" applied.`);
    }
  };
}

const theme = createDarkTheme('dark');
theme.apply();
```

```python
class DarkTheme:
    def __init__(self):
        self.colors = {
            "primary": "#10B981",
            "heading_bg": "#23e89a",
            "code_bg": "#0F172A",
        }

    def render(self, text):
        for k, v in self.colors.items():
            print(f"  {k}: {v}")

theme = DarkTheme()
theme.render("Hello Dark World")
```
